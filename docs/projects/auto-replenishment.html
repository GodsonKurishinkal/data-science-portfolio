1. Replenishment Engine Core

from dataclasses import dataclass
from typing import Dict, List, Optional
from enum import Enum
import pandas as pd
import numpy as np

class ReplenishmentPolicy(Enum):
    PERIODIC_REVIEW = "periodic_review"  # (s, S) policy
    CONTINUOUS_REVIEW = "continuous_review"  # (s, Q) policy
    MIN_MAX = "min_max"

@dataclass
class ReplenishmentOrder:
    item_id: str
    source_location: str
    dest_location: str
    quantity: int
    priority: str
    policy_used: ReplenishmentPolicy
    reason: str

class ReplenishmentEngine:
    def __init__(self, config: dict):
        self.config = config
        self.scenario = config['scenario']
        self.classifiers = self._init_classifiers()
        self.safety_stock_calc = SafetyStockCalculator(config)
        
    def generate_orders(
        self,
        inventory_data: pd.DataFrame,
        demand_forecast: pd.DataFrame
    ) -> List[ReplenishmentOrder]:
        """Generate replenishment orders for all items."""
        
        # Classify items
        classified = self._classify_items(inventory_data)
        
        # Calculate requirements
        requirements = self._calculate_requirements(
            classified, 
            demand_forecast
        )
        
        # Apply policies and generate orders
        orders = []
        for _, item in requirements.iterrows():
            policy = self._select_policy(item['classification'])
            order = self._apply_policy(item, policy)
            if order:
                orders.append(order)
        
        # Prioritize and consolidate
        orders = self._prioritize_orders(orders)
        orders = self._consolidate_orders(orders)
        
        return orders
    
    def _classify_items(self, data: pd.DataFrame) -> pd.DataFrame:
        """Apply ABC-XYZ-FMR classification."""
        
        # ABC - Value based
        data = self.classifiers['abc'].classify(data)
        
        # XYZ - Variability based
        data = self.classifiers['xyz'].classify(data)
        
        # FMR - Velocity based
        data = self.classifiers['fmr'].classify(data)
        
        # Combined classification
        data['classification'] = (
            data['abc_class'] + 
            data['xyz_class'] + 
            data['fmr_class']
        )
        
        return data
    
    def _select_policy(self, classification: str) -> ReplenishmentPolicy:
        """Select inventory policy based on classification."""
        
        policy_map = self.config.get('policy_map', {
            'AXF': ReplenishmentPolicy.CONTINUOUS_REVIEW,
            'AXM': ReplenishmentPolicy.CONTINUOUS_REVIEW,
            'AYF': ReplenishmentPolicy.PERIODIC_REVIEW,
            'BXF': ReplenishmentPolicy.PERIODIC_REVIEW,
            'CZR': ReplenishmentPolicy.MIN_MAX,
        })
        
        return policy_map.get(
            classification, 
            ReplenishmentPolicy.PERIODIC_REVIEW
        )

2. Safety Stock Calculator

from scipy import stats

class SafetyStockCalculator:
    def __init__(self, config: dict):
        self.config = config
        self.service_levels = config.get('service_levels', {
            'A': 0.98,  # High service for A items
            'B': 0.95,  # Standard service for B items
            'C': 0.90,  # Lower service for C items
        })
        
    def calculate(
        self,
        item: dict,
        method: str = 'dynamic'
    ) -> float:
        """Calculate safety stock using specified method."""
        
        if method == 'standard':
            return self._standard_safety_stock(item)
        elif method == 'dynamic':
            return self._dynamic_safety_stock(item)
        elif method == 'capacity_aware':
            return self._capacity_aware_safety_stock(item)
        else:
            raise ValueError(f"Unknown method: {method}")
    
    def _standard_safety_stock(self, item: dict) -> float:
        """Standard safety stock: Z √ó œÉ √ó ‚àöLT"""
        
        service_level = self.service_levels.get(
            item['abc_class'], 0.95
        )
        z_score = stats.norm.ppf(service_level)
        
        demand_std = item['demand_std']
        lead_time = item['lead_time_days']
        
        safety_stock = z_score * demand_std * np.sqrt(lead_time)
        
        return max(safety_stock, 0)
    
    def _dynamic_safety_stock(self, item: dict) -> float:
        """Dynamic safety stock with demand pattern adjustment."""
        
        base_ss = self._standard_safety_stock(item)
        
        # Adjust for demand trend
        trend_factor = 1.0
        if item.get('demand_trend', 0) > 0.1:
            trend_factor = 1.2  # Increasing demand
        elif item.get('demand_trend', 0) < -0.1:
            trend_factor = 0.8  # Decreasing demand
        
        # Adjust for seasonality
        season_factor = item.get('seasonality_factor', 1.0)
        
        # Adjust for lead time variability
        lt_var_factor = 1 + (item.get('lead_time_cv', 0) * 0.5)
        
        dynamic_ss = base_ss * trend_factor * season_factor * lt_var_factor
        
        return max(dynamic_ss, 0)
    
    def _capacity_aware_safety_stock(self, item: dict) -> float:
        """Safety stock constrained by storage capacity."""
        
        dynamic_ss = self._dynamic_safety_stock(item)
        
        # Get capacity constraint
        max_capacity = item.get('storage_capacity', float('inf'))
        current_inventory = item.get('current_inventory', 0)
        
        # Available capacity
        available = max_capacity - current_inventory
        
        # Constrain safety stock
        constrained_ss = min(dynamic_ss, available * 0.3)
        
        return max(constrained_ss, 0)

3. YAML Configuration Example

# config/config.yaml - DC to Store Scenario
scenario: dc_to_store
version: "1.0"

# Classification thresholds
classification:
  abc:
    A_threshold: 0.80  # Top 80% of value
    B_threshold: 0.95  # Next 15%
    # C: remaining 5%
  xyz:
    X_threshold: 0.50  # CV < 0.5 (stable)
    Y_threshold: 1.00  # CV 0.5-1.0 (variable)
    # Z: CV > 1.0 (erratic)
  fmr:
    F_threshold: 0.70  # Fast movers (top 70% velocity)
    M_threshold: 0.90  # Medium movers (next 20%)
    # R: Rare movers (bottom 10%)

# Service levels by ABC class
service_levels:
  A: 0.98
  B: 0.95
  C: 0.90

# Policy mapping by classification
policy_map:
  AXF: continuous_review
  AXM: continuous_review
  AYF: periodic_review
  AYM: periodic_review
  BXF: periodic_review
  BYF: periodic_review
  CZR: min_max
  default: periodic_review

# Lead time settings
lead_time:
  standard_days: 3
  express_days: 1
  buffer_days: 1

# Order constraints
ordering:
  min_order_quantity: 10
  max_order_quantity: 1000
  order_multiple: 5
  consolidation_window_hours: 4

# Safety stock method
safety_stock:
  method: dynamic  # standard, dynamic, capacity_aware
  min_days_cover: 3
  max_days_cover: 30

# Alert thresholds
alerts:
  critical_days_cover: 2
  warning_days_cover: 5
  excess_days_cover: 45
