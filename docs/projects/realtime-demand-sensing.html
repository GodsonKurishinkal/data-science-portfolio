1. Real-Time Demand Sensing

demand_sensor.py Python 
import pandas as pd
import numpy as np
from prophet import Prophet
from sklearn.ensemble import IsolationForest

class DemandSensor:
    def __init__(self, sku, baseline_days=90):
        self.sku = sku
        self.baseline_days = baseline_days
        self.prophet_model = Prophet(
            yearly_seasonality=True,
            weekly_seasonality=True,
            daily_seasonality=True
        )
        self.anomaly_detector = IsolationForest(
            contamination=0.1,
            random_state=42
        )
        
    def process_signal(self, real_time_data):
        """
        Process real-time sales signal and detect anomalies
        """
        # Compute hourly demand signal
        current_signal = self._compute_demand_signal(real_time_data)
        
        # Get baseline expectation
        baseline = self._get_baseline_forecast(real_time_data['timestamp'])
        
        # Calculate deviation
        deviation = (current_signal - baseline) / (baseline + 1e-6)
        
        # Detect anomaly
        is_anomaly = self._detect_anomaly(current_signal, baseline)
        
        # Classify severity
        severity = self._classify_severity(deviation, is_anomaly)
        
        return {
            'sku': self.sku,
            'timestamp': real_time_data['timestamp'],
            'current_signal': current_signal,
            'baseline': baseline,
            'deviation': deviation,
            'is_anomaly': is_anomaly,
            'severity': severity,
            'alert_required': is_anomaly and severity != 'normal'
        }
    
    def _compute_demand_signal(self, data):
        """
        Compute demand signal from streaming data
        """
        # Aggregate recent transactions
        recent_sales = data['sales_last_hour']
        inventory_velocity = data['units_sold'] / max(data['hours_in_stock'], 1)
        web_traffic = data.get('web_views', 0)
        
        # Weighted signal
        signal = (
            0.6 * recent_sales +
            0.3 * inventory_velocity * 24 +  # Daily rate
            0.1 * web_traffic * 0.05  # Conversion proxy
        )
        
        return signal
    
    def _detect_anomaly(self, current, baseline):
        """
        Detect if current signal is anomalous
        """
        # Statistical threshold
        upper_bound = baseline * 1.5
        lower_bound = baseline * 0.5
        
        if current > upper_bound or current < lower_bound:
            return True
        
        # ML-based detection
        features = np.array([[current, baseline, current - baseline]])
        prediction = self.anomaly_detector.predict(features)
        
        return prediction[0] == -1
    
    def _classify_severity(self, deviation, is_anomaly):
        """
        Classify severity of deviation
        """
        if not is_anomaly:
            return 'normal'
        
        abs_deviation = abs(deviation)
        
        if abs_deviation > 1.0:
            return 'critical'
        elif abs_deviation > 0.5:
            return 'warning'
        else:
            return 'info'

2. Automated Replenishment Engine

trigger_engine.py Python 
class ReplenishmentTrigger:
    def __init__(self, sku, lead_time_days, service_level=0.95):
        self.sku = sku
        self.lead_time = lead_time_days
        self.service_level = service_level
        self.z_score = self._get_z_score(service_level)
        
    def evaluate_trigger(self, current_inventory, demand_forecast, demand_std):
        """
        Determine if replenishment order should be triggered
        """
        # Calculate reorder point
        reorder_point = self._calculate_reorder_point(
            demand_forecast, 
            demand_std
        )
        
        # Calculate order quantity
        order_quantity = self._calculate_order_quantity(
            demand_forecast,
            demand_std,
            current_inventory
        )
        
        # Determine urgency
        days_until_stockout = self._estimate_stockout_days(
            current_inventory,
            demand_forecast
        )
        
        # Decision logic
        should_order = current_inventory <= reorder_point
        urgency = self._calculate_urgency(days_until_stockout)
        
        return {
            'sku': self.sku,
            'should_order': should_order,
            'order_quantity': order_quantity if should_order else 0,
            'reorder_point': reorder_point,
            'current_inventory': current_inventory,
            'days_until_stockout': days_until_stockout,
            'urgency': urgency,
            'method': 'expedited' if urgency == 'critical' else 'standard'
        }
    
    def _calculate_reorder_point(self, demand_forecast, demand_std):
        """
        Calculate safety stock and reorder point
        """
        # Average demand during lead time
        avg_demand = demand_forecast * self.lead_time
        
        # Safety stock
        safety_stock = self.z_score * demand_std * np.sqrt(self.lead_time)
        
        # Reorder point
        rop = avg_demand + safety_stock
        
        return max(rop, 0)
    
    def _calculate_order_quantity(self, demand_forecast, demand_std, current_inv):
        """
        Calculate optimal order quantity (Economic Order Quantity)
        """
        # Annual demand
        annual_demand = demand_forecast * 365
        
        # Simplified EOQ
        order_quantity = np.sqrt(
            (2 * annual_demand * 100) / 5  # holding cost
        )
        
        # Adjust for current inventory
        adjusted_quantity = max(order_quantity - current_inv, 0)
        
        return int(adjusted_quantity)
    
    def _estimate_stockout_days(self, current_inventory, daily_demand):
        """
        Estimate days until stockout
        """
        if daily_demand <= 0:
            return 999
        
        days = current_inventory / daily_demand
        return max(days, 0)
    
    def _calculate_urgency(self, days_until_stockout):
        """
        Calculate urgency level
        """
        if days_until_stockout < self.lead_time:
            return 'critical'
        elif days_until_stockout < self.lead_time * 1.5:
            return 'high'
        elif days_until_stockout < self.lead_time * 2:
            return 'medium'
        else:
            return 'low'
    
    def _get_z_score(self, service_level):
        """
        Get z-score for service level
        """
        from scipy import stats
        return stats.norm.ppf(service_level)
