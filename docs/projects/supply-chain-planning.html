1. Supply Chain Planner (Core Orchestrator)

from dataclasses import dataclass
from typing import Dict, List, Optional
from datetime import datetime
import pandas as pd

@dataclass
class PlanningResult:
    """Unified result from planning cycle."""
    demand: Optional['DemandResult'] = None
    inventory: Optional['InventoryResult'] = None
    pricing: Optional['PricingResult'] = None
    network: Optional['NetworkResult'] = None
    sensing: Optional['SensingResult'] = None
    replenishment: Optional['ReplenishmentResult'] = None
    kpis: Dict[str, float] = None
    timestamp: datetime = None

class SupplyChainPlanner:
    """Master orchestrator for supply chain planning."""
    
    def __init__(self, config: 'PlanningConfig'):
        self.config = config
        self.integrations = self._init_integrations()
        self.kpi_calculator = KPICalculator()
        self.alert_manager = AlertManager()
        
    def run_planning_cycle(
        self,
        planning_horizon: str = 'monthly',
        include_modules: List[str] = None
    ) -> PlanningResult:
        """
        Execute a complete planning cycle.
        
        Args:
            planning_horizon: 'monthly', 'weekly', 'daily', 'realtime'
            include_modules: List of modules to include
        """
        result = PlanningResult(timestamp=datetime.now())
        
        modules = include_modules or [
            'demand', 'inventory', 'pricing', 
            'network', 'sensing', 'replenishment'
        ]
        
        # Execute in dependency order
        if 'demand' in modules:
            result.demand = self.integrations['demand'].run_forecast(
                horizon_days=self._get_horizon_days(planning_horizon)
            )
        
        if 'inventory' in modules:
            result.inventory = self.integrations['inventory'].run_optimization(
                demand_forecast=result.demand
            )
        
        if 'pricing' in modules:
            result.pricing = self.integrations['pricing'].run_optimization(
                demand_data=result.demand
            )
        
        if 'network' in modules:
            result.network = self.integrations['network'].run_optimization()
        
        if 'sensing' in modules:
            result.sensing = self.integrations['sensing'].run_sensing()
        
        if 'replenishment' in modules:
            result.replenishment = self.integrations['replenishment'].run_replenishment(
                inventory_result=result.inventory,
                demand_result=result.demand
            )
        
        # Calculate unified KPIs
        result.kpis = self.kpi_calculator.calculate_all(result)
        
        # Generate alerts for exceptions
        self._generate_alerts(result)
        
        return result
    
    def generate_sop_plan(
        self,
        horizon_months: int = 3,
        scenarios: List[str] = None
    ) -> 'SOPPlan':
        """Generate S&OP plan with scenario analysis."""
        
        scenarios = scenarios or ['base', 'optimistic', 'pessimistic']
        
        scenario_results = {}
        for scenario in scenarios:
            scenario_results[scenario] = self._run_scenario(
                scenario, horizon_months
            )
        
        # Identify gaps and recommendations
        gaps = self._identify_gaps(scenario_results)
        recommendations = self._generate_recommendations(gaps)
        
        return SOPPlan(
            planning_date=datetime.now(),
            horizon_months=horizon_months,
            scenarios=scenario_results,
            gaps=gaps,
            recommendations=recommendations
        )
    
    def monitor_exceptions(self) -> List[Dict]:
        """Monitor for exceptions across all modules."""
        
        exceptions = []
        
        # Check sensing anomalies
        if self.integrations['sensing'].last_result:
            anomalies = self.integrations['sensing'].get_anomalies()
            for anomaly in anomalies:
                exceptions.append({
                    'type': 'demand_anomaly',
                    'severity': anomaly['severity'],
                    'item_id': anomaly['item_id'],
                    'source': 'sensing'
                })
        
        # Check inventory alerts
        if self.integrations['inventory'].last_result:
            inventory_alerts = self.integrations['inventory'].get_alerts()
            exceptions.extend(inventory_alerts)
        
        # Check replenishment issues
        if self.integrations['replenishment'].last_result:
            replen_issues = self.integrations['replenishment'].get_issues()
            exceptions.extend(replen_issues)
        
        return sorted(exceptions, key=lambda x: x.get('severity', 'LOW'))

2. Workflow Engine

from enum import Enum
from typing import Callable, Dict, List
from collections import defaultdict

class WorkflowStatus(Enum):
    PENDING = "pending"
    RUNNING = "running"
    COMPLETED = "completed"
    FAILED = "failed"

@dataclass
class WorkflowStep:
    name: str
    step_type: str
    dependencies: List[str]
    handler: Callable = None

class PlanningWorkflow:
    """Workflow definition and execution engine."""
    
    def __init__(self, workflow_type: str):
        self.workflow_type = workflow_type
        self.steps: Dict[str, WorkflowStep] = {}
        self.execution_order: List[str] = []
        self._init_default_workflow(workflow_type)
    
    def _init_default_workflow(self, workflow_type: str):
        """Initialize default workflow based on type."""
        
        workflows = {
            'monthly_sop': [
                ('demand_forecast', 'demand', []),
                ('inventory_optimization', 'inventory', ['demand_forecast']),
                ('network_optimization', 'network', ['demand_forecast']),
                ('pricing_strategy', 'pricing', ['demand_forecast']),
                ('replenishment_plan', 'replenishment', ['inventory_optimization']),
                ('kpi_calculation', 'kpi', ['demand_forecast', 'inventory_optimization']),
            ],
            'weekly': [
                ('demand_forecast', 'demand', []),
                ('inventory_adjustment', 'inventory', ['demand_forecast']),
                ('pricing_update', 'pricing', ['demand_forecast']),
                ('replenishment_orders', 'replenishment', ['inventory_adjustment']),
            ],
            'daily': [
                ('sensing_update', 'sensing', []),
                ('exception_detection', 'detection', ['sensing_update']),
                ('replenishment_execution', 'replenishment', ['exception_detection']),
                ('alert_generation', 'alerts', ['exception_detection']),
            ],
            'realtime': [
                ('anomaly_detection', 'sensing', []),
                ('exception_handling', 'exception', ['anomaly_detection']),
                ('alert_escalation', 'alerts', ['exception_handling']),
            ],
        }
        
        for name, step_type, deps in workflows.get(workflow_type, []):
            self.add_step(name, step_type, deps)
        
        self.execution_order = self._topological_sort()
    
    def _topological_sort(self) -> List[str]:
        """Sort steps by dependencies using topological sort."""
        
        in_degree = defaultdict(int)
        for step in self.steps.values():
            for dep in step.dependencies:
                in_degree[step.name] += 1
        
        queue = [name for name in self.steps if in_degree[name] == 0]
        order = []
        
        while queue:
            current = queue.pop(0)
            order.append(current)
            
            for step in self.steps.values():
                if current in step.dependencies:
                    in_degree[step.name] -= 1
                    if in_degree[step.name] == 0:
                        queue.append(step.name)
        
        return order
    
    def execute(self, context: Dict, config: 'PlanningConfig') -> 'WorkflowResult':
        """Execute workflow in dependency order."""
        
        result = WorkflowResult(
            workflow_type=self.workflow_type,
            start_time=datetime.now()
        )
        
        for step_name in self.execution_order:
            step = self.steps[step_name]
            try:
                step_result = self._execute_step(step, context, config)
                context[step_name] = step_result
                result.step_results[step_name] = step_result
            except Exception as e:
                result.status = WorkflowStatus.FAILED
                result.error = str(e)
                break
        else:
            result.status = WorkflowStatus.COMPLETED
        
        result.end_time = datetime.now()
        return result
