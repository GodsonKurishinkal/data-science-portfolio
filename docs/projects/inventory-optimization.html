Business Problem

This engine provides scientific inventory policies using Economic Order Quantity (EOQ), safety stock optimization, and ABC-XYZ segmentation to achieve the optimal balance between service level and inventory investment.

Quick Start

from src.inventory.abc_analysis import ABCAnalyzer
from src.inventory.safety_stock import SafetyStockCalculator
from src.inventory.eoq import EOQCalculator

# ABC-XYZ Classification
abc = ABCAnalyzer(df)
classified = abc.classify_items(
    value_col='annual_value',
    cv_col='demand_cv'
)

# Safety Stock Calculation
ss_calc = SafetyStockCalculator(service_level=0.98)
safety_stock = ss_calc.calculate(
    demand_std=15.5,
    lead_time=7
)

# Economic Order Quantity
eoq = EOQCalculator(
    annual_demand=10000,
    order_cost=50,
    holding_cost=2.5
)
optimal_qty = eoq.calculate()

print(f"Safety Stock: {safety_stock:.0f} units")
print(f"EOQ: {optimal_qty:.0f} units")
